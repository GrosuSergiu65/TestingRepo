import java.util.Scanner;


public class JAvaHello {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        int number1;
        int number2;
        int number3;

// Take three input numbers
        System.out.print("Enter first integer: ");
        number1 = input.nextInt();

        System.out.print("Enter second integer:");
        number2 = input.nextInt();

        System.out.print("Enter third integer:");
        number3 = input.nextInt();

// Calculate sum of numbers
        int sum;
        sum = number1 + number2 + number3;
        System.out.println(sum + " is the sum of numbers");

// Calculate average number
        System.out.println((sum/3) + " is the average number");

// Calcultate product number
        int product;
        product = number1 * number2 * number3;
        System.out.println(product + " is the product number");

// Indetify largest and smalles number
        if (number1 >= number2) {
            if (number2 >= number3)
                System.out.println(number1 + " is the largest number");
            else
                System.out.println(number3 + " is the largest number");
        } else {
            if (number2 >= number3)
                System.out.println(number2 + " is the largest number");
            else
                System.out.println(number3 + " is the largest number");
        }
        if (number1 <= number2) {
            if (number2 <= number3)
                System.out.println(number1 + " is the smallest number");
            else
                System.out.println(number3 + " is the smallest number");
        } else {
            if (number2 <= number3)
                System.out.println(number2 + " is the smallest number");
            else
                System.out.println(number3 + " is the smallest number");
        }
    }
}
